// DASO — Memory managed automatically
struct Config {
    host    str,
    port    int,
    debug   bool,
}

trait Validate {
    act is_valid(self) bool
}

impl Validate for Config {
    act is_valid(self) bool ->
        self.port > 0 && self.port < 65536
}

act make_config() Config {
    cfg := Config {
        host  "localhost",
        port  8080,
        debug false,
    }
    guard cfg.is_valid() else {
        retn Config { host "0.0.0.0", port 3000, debug true }
    }
    retn cfg
}

Why Ryx

Built for the next generation of systems programming

Ryx delivers C-level speed, Rust-like safety, and Python-like productivity in one carefully designed language.

⚡

Native Performance

Transpiles to C11–C23 and compiles with GCC/Clang for production builds. ryx run uses TCC for instant iteration loops.

🛡️

DASO Memory Safety

Deterministic Automatic Scoped Ownership. No garbage collector, no hidden pauses. O(1) region deallocation. No use-after-free, ever, in safe code.

🧠

AI/ML Native

First-class tensor type with GPU acceleration support built into the language primitives. No external bindings required.

✨

Minimal Syntax

~25 keywords total. No parentheses in control flow. Spine_Case naming convention. Expression bodies with ->. Write less, mean more.

🔗

Zero-Cost Async

First-class async/await with async pipeline operator ~>. Eliminates nested callbacks. Reads like a flowchart.

📦

Clean Module System

Every file is a module. Dot-access namespacing. Root imports with ~. Private by default, explicit with pub.

⚙️

Compiler-Centric Design

No visible lifetimes. No borrow annotations. No memory keywords for common code. The compiler performs complex reasoning so you don't have to.

🎯

Deterministic Behavior

No GC pauses. No background runtime threads. No hidden allocations. Memory is freed at exactly known, analyzable points in every program.

🔒

Safe Unsafe Escape Hatch

Need raw control? Use unsafe { } blocks — fully isolated from safe code, zero overhead, statically auditable.

02_functions.ryx
// Expression body (one-liner with ->)
act square(n int) int -> n * n
act add(a int, b int) int -> a + b

// Full body function
act greet(name str) str {
    retn "Hello, {name}! Welcome to Ryx."
}

// Generic function
act first[T](arr [T]) Option[T] {
    if arr.len() == 0 { retn None }
    retn Some(arr[0])
}

// Async function
async act fetch(url str) Result[str, Error] {
    data := http.get(url).await?
    retn Ok(data.text())
}

// Async pipeline with ~> operator
async act pipeline() {
    result := fetch("https://api.ryx.dev/data")
        ~> parse_json()
        ~> validate()
        ~> transform()

    println(result)
}

// Guard — early return for clean code
act process(user User) Result[str, str] {
    guard user.name != "" else { retn Err("Name required") }
    guard user.id > 0    else { retn Err("Invalid ID") }
    retn Ok("Processed: {user.name}")
}
03_structs.ryx
// Define a struct
struct Point {
    x float,
    y float,
}

// Enum with variants
enum Status {
    Active,
    Idle,
    Offline(str),   // variant with data
}

// Trait — shared behavior (like interfaces)
trait Describe {
    act describe(self) str
}

// Implement trait for struct
impl Describe for Point {
    act describe(self) str ->
        "Point({self.x}, {self.y})"
}

// Pattern matching on enum
act handle(status Status) {
    match status {
        Status.Active      -> println("Online"),
        Status.Idle        -> println("Away"),
        Status.Offline(msg) -> println("Gone: {msg}"),
    }
}

// Type alias
alias Vector3 = [float; 3]
alias UserMap = Map[int, User]

numbers := [1, 2, 3, 4, 5]

// Dynamic vectors
mut names := vec[str]()
names.push("Alice")

// Slices and ranges

    println("{i}: {names[i]}")

// Maps
mut scores := map[str, int]()
scores.insert("Alice", 95)
scores.insert("Bob", 87)

// Map with literal syntax
ages := map {
    "Alice"   25,
    "Bob"     30,
    "Charlie" 28,
}

// Iterate over map
loop (name, age) in ages {
    println("{name} is {age} years old")
}

// Membership check
if "Alice" in scores {
    println("Found Alice: {scores.get("Alice").unwrap()}")
}
05_errors.ryx
// Result[T, E] — explicit error handling
act divide(a float, b float) Result[float, str] {
    if b == 0.0 { retn Err("Division by zero") }
    retn Ok(a / b)
}

// ? operator — propagate error upward
act process() Result[int, Error] {
    data  := fetch_data()?      // early return on Err
    value := parse(data)?       // same here
    retn Ok(value * 2)
}

// Option[T] — nullable values without null
act find_user(id int) Option[User] {
    users := load_users()
    retn users.get(id)
}

// Pattern match on Result
act main() {
    match divide(10.0, 3.0) {
        Ok(result)  -> println("Result: {result}"),
        Err(reason) -> println("Error: {reason}"),
    }

    // unwrap_or for defaults
    user := find_user(42).unwrap_or(User { id 0, name "Guest" })

    // defer — guaranteed cleanup
    f := File.open("data.txt")?
    defer f.close()
    content := f.read()?
    println(content)
}
06_daso.ryx
// ── DASO: Deterministic Automatic Scoped Ownership ──

// Region freed at end of scope — O(1) deallocation
act compute() {
    a := vec[int]()              // in region
    b := Map[str, int]()         // in region
    c := User { id 1, name "A" } // in region
} // ← entire region freed here instantly

// Escape analysis: compiler auto-promotes escaping values
act make_user() User {
    u := User { id 1, name "Ryx" }
    retn u  // compiler detects escape → promotes to heap
}           // no annotation needed

// Move semantics — single ownership
u := User()
v := u       // u moves to v
// u is now invalid — compile error if used

// Shared — explicit reference counting (opt-in)
shared node := Node()
a := node    // ref count: 2
b := node    // ref count: 3

// Unsafe — explicit escape hatch (raw memory)
unsafe {
    buf := alloc(1024)   // raw heap
    write_data(buf)
    free(buf)            // manual cleanup
}

// Compiler rule: if it doesn't escape, you never think about memory.
// If it escapes, Ryx handles it automatically.
// If you share, you say `shared`. That's it.
DASO

Memory safety without the complexity

Deterministic Automatic Scoped Ownership — Ryx's novel memory model combines the best of regions, ownership, and escape analysis into one invisible system.

Memory Region Visualization

user: User

✓ DASO Guarantees
No GC pauses No leaks No use-after-free O(1) dealloc Zero annotations

The compiler does the hard work.
You write the logic.

🏔️

Function-Level Regions

Every act creates an implicit memory region. All allocations go in. When the function exits, the entire region is released in O(1) — one pointer reset.

🔮

Automatic Escape Analysis

If a value needs to outlive its scope, the compiler detects this and automatically promotes it to heap-backed storage with ownership. No annotations. No developer action.

🎯

Ownership Without Verbosity

Ryx enforces single ownership and safe borrows — but entirely at the compiler level. No lifetimes in syntax. No borrow checker in your face. All guarantees still hold.

⚡

Opt-In Shared Ownership

Use shared for reference-counted values. Non-atomic by default, atomic automatically when crossing async boundaries.

Language Design

~25 keywords. No more, no less.

Every keyword in Ryx exists for a purpose. No noise, no ceremony. Each one maps to a clear, semantic intent.

act retn mut const shared struct enum trait impl alias use pub if else loop break next match guard defer async await unsafe in as is self true false

Compare to Rust (~50+ keywords), C++ (90+), or Java (50+).
